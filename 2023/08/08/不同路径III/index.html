不同路径III

矩阵 DFS 回溯 数组
算法
发布日期:   2023-08-08
文章字数:   785
阅读时长:   4 分
阅读次数:  

题目描述：
在二维网格 grid 上，有 4 种类型的方格：

1 表示起始方格。且只有一个起始方格。
2 表示结束方格，且只有一个结束方格。
0 表示我们可以走过的空方格。
-1 表示我们无法跨越的障碍。
返回在四个方向（上、下、左、右）上行走时，从起始方格到结束方格的不同路径的数目。

每一个无障碍方格都要通过一次，但是一条路径中不能重复通过同一个方格。

这是一个经典的DFS题，题目要求必须通过所有的无障碍方格，所以可以以这个为切入点，写出递推公式。
当到达终点并且通过所有的空方格就是递归出口。
我们可以记一个变量emptyCells统计空方格数量。定义一个数组dir表示行走方向。

class Solution {
    int[][] directions = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    int count = 0;

    public int uniquePathsIII(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int startX = 0, startY = 0, endX = 0, endY = 0;
        int emptyCells = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    startX = i;
                    startY = j;
                } else if (grid[i][j] == 2) {
                    endX = i;
                    endY = j;
                } else if (grid[i][j] == 0) {
                    emptyCells++;
                }
            }
        }

        dfs(grid, startX, startY, endX, endY, emptyCells + 1);
        return count;
    }

    private void dfs(int[][] grid, int x, int y, int endX, int endY, int emptyCells) {
        if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length || grid[x][y] == -1) {
            return;
        }

        if (x == endX && y == endY) {
            //判断是否走完所有空方格
            if (emptyCells == 0) {
                count++;
            }
            return;
        }

        int prev = grid[x][y];
        grid[x][y] = -1;
        for (int[] dir : directions) {
            dfs(grid, x + dir[0], y + dir[1], endX, endY, emptyCells - 1);
        }
        //还原初始状态，方便其他路径访问
        grid[x][y] = prev;
    }
}

这里有一个细节，因为起始位置也算一个走过的空格，dfs(grid, startX, startY, endX, endY, emptyCells + 1);所以这里应该是emptyCells+1

class Solution {
    int[][] dir = new int[][]{{0, 1}, {0, -1}, {-1, 0}, {1, 0}};

    public int uniquePathsIII(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        boolean[][] visited = new boolean[m][n];
        int startX = 0, startY = 0, endX = 0, endY = 0;
        int emptyCells = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    startX = i;
                    startY = j;
                    visited[i][j] = true;
                } else if (grid[i][j] == 2) {
                    endX = i;
                    endY = j;
                } else if (grid[i][j] == 0) {
                    emptyCells++;
                }
            }
        }

        return dfs(grid, visited, startX, startY, endX, endY, emptyCells + 1);
    }

    public int dfs(int[][] grid, boolean[][] visited, int x0, int y0, int x, int y, int emptyCells) {
        int m = grid.length, n = grid[0].length;

        if (x0 == x && y0 == y) {
            if (emptyCells == 0) {
                return 1;
            }
            return 0;
        }

        int count = 0;
        for (int i = 0; i < dir.length; i++) {
            int newX = x0 + dir[i][0];
            int newY = y0 + dir[i][1];

            if (newX >= 0 && newX < m && newY >= 0 && newY < n && !visited[newX][newY] && grid[newX][newY] != -1) {
                visited[newX][newY] = true;
                count += dfs(grid, visited, newX, newY, x, y, emptyCells - 1);
                visited[newX][newY] = false;
            }
        }

        return count;
    }
}
